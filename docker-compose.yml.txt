version: '3.8'

services:
  db:
    image: mysql:8.0
    command: --default-authentication-plugin=mysql_native_password
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: example
      MYSQL_DATABASE: trip # Name of the database you want to create automatically
      MYSQL_USER: user # Non-root user you want to create automatically
      MYSQL_PASSWORD: password # Password for the non-root user
    ports:
      - "3306:3306" # Map port 3306 inside the container to port 3306 on the host
    volumes:
      - mysql_data:/var/lib/mysql # Persist data across container recreations

  adminer:
    image: adminer
    restart: always
    ports:
      - "8080:8080" # Access Adminer from http://localhost:8080

volumes:
  mysql_data: # Defines the volume
